- Variables of primitive types are copied by value
- Objects are copied by reference
- Object.assign({}, obj) copies object by value (Objects in Objects are copied by reference)
- Spread-Operator {...obj} copies object by value (Objects in Objects are copied by reference)
-- {...obj, property: {...obj.property}}: property is called by value
- In functions:
-- variables of primitive types are passed by value
-- objects are passed by reference
-- library lodash (lodash.com) contains many auxilary functions such as cloneDeep()
   which copies objects by value for all levels, i.e. considering objects in objects


// Pass by value
let name1 = 'mohamed';
let name2 = 'mohamed';
console.log(name1 === name2); -> true

name2 = 'Ahmed';
console.log(name1 === name2); -> false

name2 = name1;
name1 = 'Zaki';
console.log(name1 === name2); -> false

// Pass by reference

console.log({} === {}); -> false -> different memory addresses

let person1 = {
  name: 'mohamed',
  last: 'ahmed',
  skills: {
    skill1: 'HTML',
    skill2: 'CSS',
  },
};

let person2 = {};
console.log(person1 === person2); -> false

person1 = person2;
console.log(person1 === person2); -> true -> same memory addresses

const target = { a: 1, b: 2 };
const source = { b: 4, c: 5 };

const returnedTarget = Object.assign(target, source);

console.log(target);
// Expected output: Object { a: 1, b: 4, c: 5 }

console.log(returnedTarget === target);
// Expected output: true
